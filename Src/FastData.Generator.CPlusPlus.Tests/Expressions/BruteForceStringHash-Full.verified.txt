        int32_t length;
        int32_t offset;
        uint64_t hash;
        length = value.Length;
        offset = 0;
        length -= offset;
        while ((length > 0))
        {
            if ((length >= 8))
            {
                hash = (hash + ReadU64(value, offset));
                offset += 8;
                length -= 8;
            }
            else
            {
                if ((length >= 4))
                {
                    hash = (hash + ReadU32(value, offset));
                    offset += 4;
                    length -= 4;
                }
                else
                {
                    if ((length >= 2))
                    {
                        hash = (hash + ReadU16(value, offset));
                        offset += 2;
                        length -= 2;
                    }
                    else
                    {
                        hash = (hash + ReadU8(value, offset));
                        offset += 1;
                        length -= 1;
                    }
                }
            }
        }

        hash = hash;
