int length;
int offset;
ulong hash;
length = input;
offset = 0;
length -= offset;
while ((length > 0))
{
    if ((length >= 8))
    {
        hash = (hash + ReadU64(input, offset));
        offset += 8;
        length -= 8;
    }
    else
    {
        if ((length >= 4))
        {
            hash = (hash + ReadU32(input, offset));
            offset += 4;
            length -= 4;
        }
        else
        {
            if ((length >= 2))
            {
                hash = (hash + ReadU16(input, offset));
                offset += 2;
                length -= 2;
            }
            else
            {
                hash = (hash + ReadU8(input, offset));
                offset += 1;
                length -= 1;
            }
        }
    }
}

hash = hash;
